// Development Firestore Rules - More permissive for localhost
// Copy these to Firebase Console > Firestore Database > Rules
// (Not Realtime Database - make sure you're in the right section!)

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow read/write access for authenticated users
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
    
    // Allow users to read/write their own data
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Allow public read for some collections (if needed)
    match /classes/{classId} {
      allow read, write: if request.auth != null;
    }
    
    // Allow attendance data access for teachers and students
    match /attendance/{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
